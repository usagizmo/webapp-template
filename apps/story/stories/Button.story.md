```svelte
<Button>Button</Button>
//=> <button type="button" />
```

```svelte
<Button type="submit">Button</Button>
//=> <button type="submit" />
```

```svelte
<Button href="#">Link</Button>
//=> <a href="#" />
```

```svelte
<Button href="#" blank>Link</Button>
//=> <a href="#" target="blank" rel="noopener noreferrer" />
```
